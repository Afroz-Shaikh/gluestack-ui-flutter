import 'package:gluestack_ui/gluestack_ui.dart';

GSButton(
  action: GSActions.primary,
  variant: GSVariants.solid,
  size: GSSize.$md,
  isDisabled: false,
  isFocusVisible: false,
  onPressed: () {},
  child: const GSHStack(
    children: [
      GSButtonText(text: "Add"),
      GSButtonIcon(icon: Icons.add)
    ],
  ),
)
