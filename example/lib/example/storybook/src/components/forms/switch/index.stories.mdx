```dart
import 'package:gluestack_ui/gluestack_ui.dart';

GSSwitch(
  size: GSSizes.\$md,
  isDisabled: false,
  style: GSStyle
  (
    checked: GSStyle
    (
      thumbColor: Colors.grey,
      activeThumbColor: Colors.blue,
    ),
  ),
  value: val1,
  onToggle: (bool value)
  {
    setState(()
    {
      val1 = value;
    });
  },
  )
```
