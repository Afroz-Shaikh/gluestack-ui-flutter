```dart
import 'package:gluestack_ui/gluestack_ui.dart';

GSAvatar(
  radius: GSBorderRadius.\$full,
  size: GSSizes.\$md,
  style: GSStyle(
    bg: Colors.orange,
    textStyle: const TextStyle(color: Colors.white),
  ),
  fallBackText: const GSAvatarFallBackText(
    'Geeky Stack',
  ),
)
```
